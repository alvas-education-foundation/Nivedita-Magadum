Largest element smaller than current element on left for every element in Array
import java.util.*; 
  
class GFG{ 
   
 
static void findMaximumBefore(int arr[], 
                         int n){ 
       
    
    for (int i = 0; i < n; i++) { 
   
        int currAns = -1; 
            
        
        for (int j = i - 1; j >= 0; j--) { 
            if (arr[j] > currAns && 
                   arr[j] < arr[i]) { 
                currAns = arr[j]; 
            } 
        } 
        System.out.print(currAns+ " "); 
    } 
} 
   
// Driver Code 
public static void main(String[] args) 
{ 
    int arr[] = { 4, 7, 6, 8, 5 }; 
   
    int n = arr.length; 
   
    // Function Call 
    findMaximumBefore(arr, n); 
} 
}
